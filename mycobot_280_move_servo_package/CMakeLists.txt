cmake_minimum_required(VERSION 3.8)
project(mycobot_280_move_servo_package)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(moveit_servo REQUIRED)
find_package(rclcpp REQUIRED)
find_package(control_msgs REQUIRED)
find_package(geometry_msgs REQUIRED)
#find_package(moveit_ros_planning_interface REQUIRED)  # 添加这一行
# uncomment the following section in order to fill in
# further dependencies manually.

add_executable(servo_keyboard_input src/servo_keyboard_input.cpp)
ament_target_dependencies(servo_keyboard_input 
rclcpp 
control_msgs 
geometry_msgs)  # 添加所需的依赖

add_executable(servo_node_main src/servo_node_main.cpp)
ament_target_dependencies(servo_node_main 
rclcpp 
moveit_servo)  # 添加所需的依赖

add_executable(servo_arm_node_main src/servo_arm_node_main.cpp)
ament_target_dependencies(servo_arm_node_main 
rclcpp 
moveit_servo)


add_executable(servo_gripper_node_main src/servo_gripper_node_main.cpp)
ament_target_dependencies(servo_gripper_node_main 
rclcpp 
moveit_servo)



add_executable(collision_object_node src/collision_object_node.cpp)
ament_target_dependencies(collision_object_node
  rclcpp
  moveit_msgs
  geometry_msgs
  shape_msgs
)

install(TARGETS
  servo_node_main 
  servo_keyboard_input
  collision_object_node
  servo_arm_node_main
  servo_gripper_node_main
  DESTINATION lib/${PROJECT_NAME}

install (
  DIRECTORY config src
  DESTINATION share/${PROJECT_NAME}
)

  
if(BUILD_TESTING)
  find_package(ament_lint_auto REQUIRED)
  # the following line skips the linter which checks for copyrights
  # comment the line when a copyright and license is added to all source files
  set(ament_cmake_copyright_FOUND TRUE)
  # the following line skips cpplint (only works in a git repo)
  # comment the line when this package is in a git repo and when
  # a copyright and license is added to all source files
  set(ament_cmake_cpplint_FOUND TRUE)
  ament_lint_auto_find_test_dependencies()
endif()
